\section{Algebra. Functions. Composition.}
\question Let $f(x) = -2+\sqrt{x+3}$ and let $g(x) = \frac1x$. Find the composition $(f\circ g)(x)$.
\begin{multicols}3
\begin{choices}
	\choice $\frac1{-2+\sqrt{x+3}}$
	\choice $-2+\sqrt{x+3}$
	\choice $-2+\frac1{\sqrt{x+3}}$
	\choice $-2+\frac1{\sqrt x+3}$
	\correctchoice $-2+\sqrt{\frac1x+3}$
	\end{choices}
	\end{multicols}
