\newcommand\content[4]{
	\par\bigskip\bigskip
	\question
	\(\begin{matrix}\begin{tikzpicture}[scale=1, very thick]
		%% #2 = 0 (f above xaxis) or 1 (f below xaxis)
		\draw[->] (-1,-1+2*#2) -- ++(right:2);
		\draw[->] (0,-1.3) -- ++(up:2.6);
		%% #3 = 0 (f increasing) or 1 (f decreasing)
		%% #4 = 0 (f concave up) or 1 (f concave down)
		\draw ({-0.7*(1-#4*2)*(1-#3*2)}, {-0.7*(1-#4*2)}) arc
		[start angle=-75*(1-#3*2), end angle=-15*(1-#3*2),
			radius={2cm*(1-#4*2)*(1-#3*2)}]
		node[right, pos={0.5+0.4*(1-#4*2)*(1-#3*2)}] {$f$};
		\end{tikzpicture}\end{matrix}\)
	\hfill
	%% #1 = black (example 1) or white (not example 1)
	\(\begin{matrix}
		\rowheader{16mm}{Above\\[-1ex] $x$-axis? \\ Below?} &
		f \fillin[\color{#1}$>0$]\\[2.5ex]
		\rowheader{25mm}{Increasing? \\[-0.5ex] Decreasing?} &
		f \fillin[\color{#1}$\searrow$] &\bc& \g f' \fillin[\color{#1}$<0$]\\[2.5ex]
		\rowheader{29mm}{Concavity?} &
		f \fillin[\color{#1}$\cup$] &\bc& \g f' \fillin[\color{#1}$\nearrow$] &\bc& \g f'' \fillin[\color{#1}$>0$]
		\end{matrix}\)
	}
